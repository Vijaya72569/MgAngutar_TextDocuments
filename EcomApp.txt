cart.component.ts

import { Component, OnInit } from '@angular/core';
import { ProductService } from '../product.service';

@Component({
  selector: 'app-cort',
  templateUrl: './cort.component.html',
  styleUrls: ['./cort.component.css']
})
export class CortComponent implements OnInit {
  selectedProduct: any = null;
  counter:number=1;
  errorMessage:string='';
  constructor(private productService: ProductService) { }

  ngOnInit(): void {
    this.productService.selectedProduct$.subscribe((product) => {
      this.selectedProduct = product;
    });

}
 
  fun1(){
 this.counter+=1;
  }
  fun2(){
    this.counter-=1;
  } 
  fun3():void{
  if(this.counter<10){ 
 this.counter+=1;
 this.errorMessage='';
  }
 else{
 this.errorMessage='you can add upto 10 products in one order'
 }
  }
  fun4():void{
  if(this.counter>1){
  
    this.counter-=1;
    this.errorMessage='';
  }
     } 
     removeProduct(): void {
      this.selectedProduct = null; // Remove the selected product
      this.errorMessage = ''; // Clear any error message if present
      this.counter = 1; // Reset counter if needed
      
    }   
}
